{

    {

        {

        }

        {

        {

            variable.CreateAsset ($"Assets/$NAMESPACE_NAME$/Data/VariableSetter/$TYPE_NAME$/{sceneName}/{name.ReplaceByRegex("[^a-zA-Z ]", string.Empty)}$TYPE_NAME$.asset");
        }
    }
}
